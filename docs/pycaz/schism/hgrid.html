157

172        """
173        try:
174            _necessary_fields = ['nodes', 'elems', 'nodes']
175            _necessary_fields_found = [i in self for i in _necessary_fields]

178            Exception(f'Necessary fields {_necessary_fields} not found!')
179        
180        # All element is in present
181        nSCHISM_hgrid_node = np.arange(self['nnode']) + 1
182        nSCHISM_hgrid_face = np.arange(self['nelem']) + 1
183        nMaxSCHISM_hgrid_face_nodes = np.arange(4)

185        ELEM_FILL_VALUE = self['elem_FillValue']

188            data=self.x, # first column

191                'nSCHISM_hgrid_node':nSCHISM_hgrid_node
192            }
193        )
194
195        SCHISM_hgrid_node_y = xr.DataArray(
196            data=self.y, # second column
197            dims=['nSCHISM_hgrid_node'],
198            coords={
199                'nSCHISM_hgrid_node':nSCHISM_hgrid_node
200            }
201        )
202
203        SCHISM_hgrid_face_nodes = xr.DataArray(
204            data=self['elems'],
205            dims=['nSCHISM_hgrid_face', 'nMaxSCHISM_hgrid_face_nodes'],
206            coords={
207                'nSCHISM_hgrid_face':nSCHISM_hgrid_face,
208                'nMaxSCHISM_hgrid_face_nodes':nMaxSCHISM_hgrid_face_nodes
209            }
210        )
211
212        if nData == 1:
213            vardata = xr.DataArray(
214                data=self.data.flatten(),
215                dims=['nSCHISM_hgrid_node'],
216                coords={
217                    'nSCHISM_hgrid_node':nSCHISM_hgrid_node
218                },
219                attrs={
220                    'header':self['header']
221                }
222            )
223        elif nData > 1:
224            varcoord = np.arange(nData)
225            vardata = xr.DataArray(
226                data=self.data,
227                dims=['nSCHISM_hgrid_node', varname],
228                coords={
229                    'nSCHISM_hgrid_node':nSCHISM_hgrid_node,
230                    varname:varcoord
231                },
232                attrs={
233                    'header':self['header']
234                }
235            )

237            Exception('Data in Gr3 is not correct! Data size mismatch.')
238
239
240        ds = xr.Dataset(
241            data_vars={
242                'SCHISM_hgrid_node_x':SCHISM_hgrid_node_x,
243                'SCHISM_hgrid_node_y':SCHISM_hgrid_node_y,
244                'SCHISM_hgrid_face_nodes':SCHISM_hgrid_face_nodes,
245                varname:vardata
246            },
247            attrs={
248                'header':self['header']
249            }
250        )
251
252        ds.SCHISM_hgrid_face_nodes.encoding['_FillValue'] = ELEM_FILL_VALUE
253
254        return ds

458def _hgrid_find_chunks(fname: str):
459    """
460    Find different chunk of the Gr3/Hgrid file
461    
462    returns: dict(header, elem, nodes, [elems, [boundaries]])
463
464    """
465    with open(fname) as f:
466        _ds = f.readlines()
467        _length = len(_ds)
468
469        # first check if there is minimum 2 lines to give us basic information
470        try:
471            assert(_length > 2)
472        except AssertionError:
473            raise Exception('Invalid length of file!')
474        
475        # Get the grid name at the first line, could be empty
476        _name = ' '.join(_ds[0].split())
477        
478        # Get the element and node counts from the second line
479        _nelem, _nnode = _ds[1].split()
480        _nelem = int(_nelem)
481        _nnode = int(_nnode)
482
483        _return_chunks = {
484            'header':_name,
485            'nelem':_nelem,
486            'nnode':_nnode
487        }
488        
489        # Try reading the nodes sagment
490        try:
491            _nodes = _ds[2:_nnode+2]
492        except IndexError:
493            raise IndexError(f'Could not read {_nnode} nodes.')
494        else:
495            _return_chunks['nodes'] = _nodes
496        
497        # If we do not hit empty line just after nodes, then try reading the element sagment
498        if _ds[_nnode+2].strip():
499            try:
500                _elems = _ds[_nnode+2:_nnode+_nelem+2]
501            except IndexError:
502                raise IndexError(f'Could not read {_nelem} elements.')
503            else:
504                _return_chunks['elems'] = _elems
505            
506        # If we do not hit empty line just after elements, then try reading the boundary sagment
507        if _ds[_nnode+_nelem+2].strip():
508            _boundaries = _ds[_nnode+_nelem+2:_length]
509            _readflag = (True, True, True)
510            _return_chunks['boundaries'] = _boundaries
511                
512    return _return_chunks

514def _hgrid_parse_nodes(chunk: list) -> dict:
515    """
516    Parse nodes from the chunk found from _find_hgrid_chunks()
517
518    returns: dict(nodes, data)
519    """
520    try:
521        _nodes = np.genfromtxt(chunk)
522    except:
523        raise Exception('Problem with parsing nodes. Check the output of _find_hgrid_chunks().')
524    else:
525        return {
526            'nodes':_nodes
527        }

529def _hgrid_parse_elements(chunk: list) -> dict:
530    """
531    Parse elements from the chunk found from _find_hgrid_chunks()
532
533    return: dict(elemtype, data)
534    """
535    # Element table can be of 3 or 4 nodes, needs parsing line by line
536    _nelem = len(chunk)
537    _elemtype = np.zeros(_nelem, dtype=int)
538    _elem_FillValue = -99999
539    _elems = np.ones((_nelem, 4), dtype=int)*_elem_FillValue
540
541    for i, _line in enumerate(chunk):
542        _columns = [int(i) for i in _line.split()]
543        _elemtype[i] = _columns[1] # i3 or 4 indicator
544        _elems[i, :_columns[1]] = _columns[2:]
545
546    return {
547        'elemtype':_elemtype,
548        'elems':_elems,
549        'elem_FillValue':_elem_FillValue
550    }

552def _hgrid_parse_boundaries(chunk: list) -> dict:
553    """
554    Parse boundaries from the chunk found from _find_hgrid_chunks()
555
556    return: dict(open_bnds, land_bnds)
557    """
558    # First the open boundaries
559    _nopen = int(chunk[0].split('=')[0])
560    _nopennodes = int(chunk[1].split('=')[0])
561    _lnum = 2 # move cursor to first line in the loop
562    _open = {}
563    for _n in np.arange(_nopen):
564        _nnodes =  int(chunk[_lnum].split('=')[0])
565        _lnum = _lnum + 1
566        _nodes = np.genfromtxt(chunk[_lnum:_lnum+_nnodes], dtype=int)
567        _open.update({_n+1:OpenBoundary(name=_n+1, nodes=_nodes)})
568        _lnum = _lnum+_nnodes # move cursor to next sagment
569
570    # Then the land boundaries
571    _nland = int(chunk[_lnum].split('=')[0])
572    _lnum = _lnum + 1
573    _nlandnodes = int(chunk[_lnum].split('=')[0])
574    _lnum = _lnum + 1
575    _land = {}
576    for _n in np.arange(_nland):
577        _nnodes, _bndtype = [int(i) for i in chunk[_lnum].split('=')[0].split()]
578        _lnum = _lnum + 1
579        _nodes = np.genfromtxt(chunk[_lnum:_lnum+_nnodes], dtype=int)
580        _land.update({_n+1:LandBoundary(name=_n+1, bndtype=_bndtype, nodes=_nodes)})
581        _lnum = _lnum+_nnodes # move cursor to next sagment
582
583    _land.keys()
584
585    return {
586        'open_bnds':_open,
587        'land_bnds':_land
588    }

591def read_gr3(fname: str) -> Gr3:
592    """
593    Reads a gr3 file and return a Gr3 dict object.
594
595    Does not through error if elems are missing.
596    """
597    chunks = _hgrid_find_chunks(fname)
598    
599    # Name, elem/node count, and nodes are by design available
600    gr3 = Gr3(header=chunks['header'], nelem=chunks['nelem'], nnode=chunks['nnode'])
601    gr3.update(_hgrid_parse_nodes(chunks['nodes']))
602
603    # Add the element table only if element is available
604    if 'elems' in chunks:
605        gr3.update(_hgrid_parse_elements(chunks['elems']))
606
607    return gr3

609def read_hgrid(fname: str) -> Hgrid:
610    """
611    Reads a hgrid file
612    """
613    chunks = _hgrid_find_chunks(fname)
614
615    try:
616        _necessary_fields = ['header', 'nelem', 'nnode', 'nodes', 'elems']
617        _necessary_fields_found = [i in chunks for i in _necessary_fields]
618        assert(np.all(_necessary_fields_found))
619    except AssertionError:
620        Exception('hgrid file does not have all the  necessary fields. Currently {_necessary_fields} are mandatory.')
621    else:
622        hgrid = Hgrid(header=chunks['header'], nelem=chunks['nelem'], nnode=chunks['nnode'])
623        hgrid.update(_hgrid_parse_nodes(chunks['nodes']))
624        hgrid.update(_hgrid_parse_elements(chunks['elems']))
625
626    try:
627        assert('boundaries' in chunks)
628    except AssertionError:
629        warnings.warn('Boundaries are not found in the file! Will set to empty values')
630        boundaries = {
631            'open_bnds':{},
632            'land_bnds':{}
633        }
634    else:
635        boundaries = _hgrid_parse_boundaries(chunks['boundaries'])
636
637    # Add nodes x,y location in the boundaries
638    for open_bnd in boundaries['open_bnds']:
639        boundaries['open_bnds'][open_bnd]['xy'] = hgrid.subset_nodes(boundaries['open_bnds'][open_bnd]['nodes'])
640
641    for land_bnd in boundaries['land_bnds']:
642        boundaries['land_bnds'][land_bnd]['xy'] = hgrid.subset_nodes(boundaries['land_bnds'][land_bnd]['nodes'])
643
644    # Update the hgrid structure
645    hgrid.update(boundaries)
646
647    return(hgrid)
